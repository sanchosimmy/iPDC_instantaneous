84 int isNumber(char *s)
85 {
86  int dot=0, num=0, end=0;
87 
88  while(isspace(*s))s++;
89  if(! *s)return 0;
90  while(*s)
91  if(isspace(*s))
92  end=1, s++;
93  else if(end)
94  return 0;
95  else if(isdigit(*s))
96  num=1, s++;
97  else if(*s=='.')
98  if(dot)
99  return 0;
100  else
101  dot=1,s++;
102  else
103  return 0;
104  return num;
105 };
