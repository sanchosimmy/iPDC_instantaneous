53 void free_cfgframe_object(struct cfg_frame *cfg) {
54 
55  int j = 0;
56  unsigned int phn,ann,dgn,num_pmu;
57  struct dgnames *t_dgnames,*r_dgnames;
58 
59  num_pmu = to_intconvertor(cfg->num_pmu);
60 
61  while(j<num_pmu) {
62 
63  free(cfg->pmu[j]->stn);
64  free(cfg->pmu[j]->idcode);
65  free(cfg->pmu[j]->data_format);
66  free(cfg->pmu[j]->fmt);
67 
68  // Extract PHNMR, DGNMR, ANNMR
69  phn = to_intconvertor(cfg->pmu[j]->phnmr);
70  ann = to_intconvertor(cfg->pmu[j]->annmr);
71  dgn = to_intconvertor(cfg->pmu[j]->dgnmr);
72 
73  if(phn != 0)
74  free_2darray(cfg->pmu[j]->cnext->phnames,phn);
75  if(ann != 0)
76  free_2darray(cfg->pmu[j]->cnext->angnames,ann);
77 
78  if(dgn != 0) {
79  t_dgnames = cfg->pmu[j]->cnext->first;
80 
81  while(t_dgnames != NULL) {
82 
83  r_dgnames = t_dgnames->dg_next;
84  free_2darray(t_dgnames->dgn,16);
85  t_dgnames = r_dgnames;
86  }
87  }
88 
89  if(phn != 0)
90  free_2darray(cfg->pmu[j]->phunit,phn);
91  if(ann != 0)
92  free_2darray(cfg->pmu[j]->anunit,ann);
93  if(dgn != 0)
94  free_2darray(cfg->pmu[j]->dgunit,dgn);
95 
96  free(cfg->pmu[j]->phnmr);
97  free(cfg->pmu[j]->annmr);
98  free(cfg->pmu[j]->dgnmr);
99  free(cfg->pmu[j]->fnom);
100  free(cfg->pmu[j]->cfg_cnt);
101 
102  j++;
103  } // End of While
104 
105  free(cfg->framesize);
106  free(cfg->idcode);
107  free(cfg->soc);
108  free(cfg->fracsec);
109  free(cfg->time_base);
110  free(cfg->data_rate);
111  free(cfg->num_pmu);
112  free(cfg);
113 }
